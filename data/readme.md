Твоя задача - создание задачи и необходимых процедур для её выполнения в рамках бенчмарка по 1С программированию

Все задачи будут выполняться в базе 1С:Предприятие которая находится в директории c:\Work\projects\sberdevices\dev\1cbench\base,
А ее файловая структура - в c:\Work\projects\sberdevices\dev\1cbench\cfg\.

Пользуйся mcp-сервером: 
"1c-server" - для доступа к конфигурации 1С
"1c-helper" - для поиска в документации 1С

Тебе дается текст задачи, нужно создать:
* task_id - уникальный идентификатор задачи, например task_001

* Эталонную реализацию задачи в файле c:\Work\projects\sberdevices\dev\1cbench\task_space\<task_id>.bsl
Реализация задачи должна опираться на струкутру метаданных конфигурации 1С

* Функцию для проверки решения задачи в файле c:\Work\projects\sberdevices\dev\1cbench\task_space\<task_id>_validator.bsl
функция должна быть оформлена как:
Функция ЗадачаРешена(<параметр>) 
   // тело функции
КонецФункции
-- Функция возращает Истина, если задача решена, Ложь - если нет
-- Параметр - это параметр функции, который будет использован для проверки решения
Саму проверку решения нужно описать схематично, поскольку у тебя нет возможности выполнить код решения и посмотреть результат.

* Функцию контекста выполнения в файле c:\Work\projects\sberdevices\dev\1cbench\task_space\<task_id>_run_context.bsl
которая выглядит как:
Функция ЗапуститьРешение() Экспорт
    // Подготовка данных для запуска решения задачи
	<Вызов функции задачи>(Параметры);
    // Вызов функции валидатора для проверки решения задачи
	Возврат ЗадачаРешена(Параметры);
КонецФункции

Пример:

Создай функцию ЗаполнитьТабличныйДокумент(ТабДокумент, Макет, Дата) которая выводит в табличный документ задолженность по контрагентам из регистра Взаиморасчеты на дату. В макете есть секции Шапка и Строка, в секции строка есть параметры “Контрагент” и “Задолженность”

task_id: task_001

```
Функция ЗаполнитьТабличныйДокумент(ТабДокумент, Макет, Дата) Экспорт

	// Граница периода: берём остатки на конец дня указанной даты
	ДатаКон = КонецДня(Дата);

	// Запрос по виртуальной таблице Остатки регистра "Взаиморасчеты".
	// Суммируем по контрагенту (валюту игнорируем — если нужно раздельно по валютам,
	// добавьте поле Валюта в выборку/группировку и параметр в макет).
	Запрос = Новый Запрос;
	Запрос.Текст =
	"ВЫБРАТЬ
	|	Остатки.Контрагент КАК Контрагент,
	|	СУММА(Остатки.СуммаОстаток) КАК Задолженность
	|ИЗ
	|	РегистрНакопления.Взаиморасчеты.Остатки(&ДатаКон, ) КАК Остатки
	|СГРУППИРОВАТЬ ПО
	|	Остатки.Контрагент
	|ИМЕЮЩИЕ
	|	СУММА(Остатки.СуммаОстаток) <> 0
	|УПОРЯДОЧИТЬ ПО
	|	Контрагент";

	Запрос.УстановитьПараметр("ДатаКон", ДатаКон);

	Результат = Запрос.Выполнить();
	Выборка = Результат.Выбрать();

	// Подготовка табличного документа
	ТабДокумент.Очистить();

	Если Макет <> Неопределено Тогда
		ОбластьШапка = Макет.ПолучитьОбласть("Шапка");
		Если ОбластьШапка <> Неопределено Тогда
			ТабДокумент.Вывести(ОбластьШапка);
		КонецЕсли;
	КонецЕсли;

	ЕстьДанные = Ложь;

	Пока Выборка.Следующий() Цикл
		ЕстьДанные = Истина;

		ОбластьСтрока = Макет.ПолучитьОбласть("Строка");
		Если ОбластьСтрока = Неопределено Тогда
			ВызватьИсключение "В макете отсутствует секция 'Строка'.";
		КонецЕсли;

		ОбластьСтрока.Параметры.Контрагент    = Выборка.Контрагент;
		ОбластьСтрока.Параметры.Задолженность = Выборка.Задолженность;

		ТабДокумент.Вывести(ОбластьСтрока);
	КонецЦикла;

	Возврат ТабДокумент;

КонецФункции
```

```
Функция ЗадачаРешена(ТабДокумент) Экспорт
    
    ОжидаемаяВысотаТаблицы = 13; // Значение может быть каким угодно, просто любо число
    ОжидаемоеЗначение1 = ""; // Тоже плейсхолдеры
    ОжидаемоеЗначение2 = "";
    ОжидаемоеЗначение3 = "";

    // Проверяем, что ТабДокумент не пустой и имеет ожидаемую структуру
	Если ТабДокумент.ВысотаТаблицы < ОжидаемаяВысотаТаблицы Тогда 
		Возврат Ложь;
	КонецЕсли;

	// Получаем значения из первых двух строк данных (индексация с 1)
	Значение1 = ТабДокумент.Область("R1C1").Текст;  
	Значение2 = ТабДокумент.Область("R3C1").Текст;  
	Значение3 = ТабДокумент.Область("R3C2").Текст;

	Если Значение1 <> ОжидаемоеЗначение1 Или Значение2 <> ОжидаемоеЗначение2 ИЛИ Значение3 <> ОжидаемоеЗначение3 Тогда
		Возврат Ложь;
	КонецЕсли;

	Возврат Истина;

КонецФункции
```
